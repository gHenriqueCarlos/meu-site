// Minimal API · C# 12
using Microsoft.AspNetCore.Builder;
using Microsoft.AspNetCore.Http;
using Microsoft.Extensions.Hosting;

var app = WebApplication.CreateBuilder(args).Build();

app.MapGet("/", () => new Dev("Carlos", "C# / Backend").Hello());
app.MapGet("/score/{n:int}", (int n) =>
    n switch
    {
        < 0     => Results.BadRequest("valor inválido"),
        >= 100  => Results.Ok("Perfeito!"),
        _         => Results.Ok(new { n, rating = Math.Round(n / 10m, 1) })
    });

app.MapGet("/health", () => Results.Ok("healthy"));
app.Run();

public record Dev(string Name, string Focus)
{
                    public string Hello() => $"Olá, {Name}! Foco: {Focus}";
}
  

// .NET MAUI · MainPage em C#
using Microsoft.Maui.Controls;

public partial class MainPage : ContentPage
{
                    int _count;
                    Label _lbl = new Label{ Text = "Cliques: 0", HorizontalOptions = LayoutOptions.Center };

                    public MainPage()
    {
        Title = "Olá, MAUI!";
        Content = new VerticalStackLayout{
            Padding = 24,
            Children =
            {
                    new Label{ Text = "Bem-vindo 👋", FontSize = 24, HorizontalOptions=LayoutOptions.Center },
                    new Button{ Text = "Contar", Command = new Command(() => { _count++; _lbl.Text = $"Cliques: {_count}"; }) },
                _lbl
            }
        };
    }
}
